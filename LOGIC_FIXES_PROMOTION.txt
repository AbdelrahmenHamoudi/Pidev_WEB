CAS B : Validation du type Utilisateur (Narrowing)

AVANT :
--------------------------------------------------
$user = $this->getUser();
if (!$user) {
    return $this->json(['success' => false, 'error' => 'Vous devez être connecté'], 401);
}
// PHPStan signale que $user (UserInterface) n'est pas forcément une entité Users.
--------------------------------------------------

APRÈS :
--------------------------------------------------
$user = $this->getUser();
if (!$user instanceof \App\Entity\Users) {
    return $this->json(['success' => false, 'error' => 'Vous devez être connecté avec un compte valide'], 401);
}
// Le type est maintenant garanti comme étant l'entité Users attendue.
--------------------------------------------------

CAS C : Suppression de code redondant (isset)

AVANT :
--------------------------------------------------
#[ORM\PrePersist]
public function onPrePersist(): void
{
    if (!isset($this->createdAt)) {
        $this->createdAt = new \DateTime();
    }
}
--------------------------------------------------

APRÈS :
--------------------------------------------------
#[ORM\PrePersist]
public function onPrePersist(): void
{
    // createdAt est déjà initialisé dans le constructeur
}
--------------------------------------------------
